    📝 技术设计

    ### 1. Agent 轮询机制

    ```javascript

    // agent-poller.js

    class AgentPoller {

    constructor() {

    this.pollInterval = 5 * 60 * 1000; // 5 分钟

    this.agents = ['xiaoyun', 'xiaoyun-judge', 'xiaoyun-recorder', 'xiaoyun-dev', 'xiaoyun-novel'];

    }

    async start() {

    // 定时轮询

    setInterval(() => this.pollAllAgents(), this.pollInterval);

    }

    async pollAllAgents() {

    for (const agent of this.agents) {

    await this.pollAgent(agent);

    }

    }

    async pollAgent(agentId) {

    // 查询分配给该 Agent 的任务

    const tasks = await this.getTasksForAgent(agentId);

    // 检查是否有可领取的任务

    const availableTasks = tasks.filter(t => t.status === 'AVAILABLE');

    if (availableTasks.length > 0) {

    // 自动领取任务

    await this.claimTask(availableTasks[0].task_id, agentId);

    // 发送通知

    await this.notifyTaskAssigned(agentId, availableTasks[0]);

    }

    }

    }

    ```

    ### 2. 任务通知机制

    ```javascript

    // notification-service.js

    class NotificationService {

    async notifyTaskAssigned(agentId, task) {

    await this.send(agentId, {

    type: 'TASK_ASSIGNED',

    task: task,

    timestamp: new Date()

    });

    }

    async notifyStatusChanged(taskId, oldStatus, newStatus) {

    await this.broadcast({

    type: 'STATUS_CHANGED',

    task_id: taskId,

    old_status: oldStatus,

    new_status: newStatus,

    timestamp: new Date()

    });

    }

    async notifyApprovalResult(taskId, result) {

    await this.send(task.requester, {

    type: 'APPROVAL_RESULT',

    task_id: taskId,

    result: result,

    timestamp: new Date()

    });

    }

    }

    ```
