🔧 服务配置

Nginx 配置

位置: /etc/nginx/conf.d/fileserver.conf

server {
    server_name yun.webyoung.cn;
    
    # 任务公告板
    location /task-board/ {
        proxy_pass http://127.0.0.1:3000/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
    
    # 文件服务器
    location / {
        proxy_pass http://127.0.0.1:8888;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
    
    # SSL 配置
    listen 443 ssl;
    ssl_certificate /etc/letsencrypt/live/yun.webyoung.cn/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/yun.webyoung.cn/privkey.pem;
}

Systemd 服务

文件服务器

[Unit]
Description=小云文件服务器
After=network.target
[Service]
Type=simple
User=admin
WorkingDirectory=/home/admin/fileserver
ExecStart=/usr/bin/python3 server.py
Restart=always
[Install]
WantedBy=multi-user.target

任务公告板

[Unit]
Description=小云任务管理系统
After=network.target
[Service]
Type=simple
User=admin
WorkingDirectory=/home/admin/.openclaw/workspace/task-board
ExecStart=/usr/bin/node src/server.js
Restart=always
Environment=NODE_ENV=production
[Install]
WantedBy=multi-user.target
