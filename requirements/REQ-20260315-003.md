#### 2.2 认领核心逻辑

```python
@app.post("/api/tasks/{task_id}/claim")
async def claim_task(task_id: str, claim_request: ClaimRequest):
    """
    认领任务 - 使用数据库事务锁防止并发冲突
    """
    async with get_db_connection() as conn:
        async with conn.transaction():
            # 1. 查询任务 (加行锁)
            task = await conn.fetchrow(
                """SELECT * FROM tasks 
                   WHERE id = $1 FOR UPDATE""",
                task_id
            )
            
            if not task:
                return {
                    "success": False,
                    "error": "任务不存在"
                }
            
            # 2. 检查是否可认领
            if task['status'] != 'AVAILABLE':
                return {
                    "success": False,
                    "error": f"任务已被认领 (状态：{task['status']})"
                }
            
            # 3. 检查认领者是否存在
            agent = await conn.fetchrow(
                "SELECT * FROM agents WHERE id = $1",
                claim_request.agent_id
            )
            
            if not agent:
                return {
                    "success": False,
                    "error": "Agent 不存在"
                }
            
            # 4. 更新任务状态
            await conn.execute(
                """UPDATE tasks 
                   SET claimed_by = $1, 
                       claimed_at = $2,
                       status = 'ASSIGNED',
                       updated_at = $3
                   WHERE id = $4""",
                claim_request.agent_id,
                datetime.now(),
                datetime.now(),
                task_id
            )
            
            # 5. 更新 Agent 状态
            await conn.execute(
                """UPDATE agents 
                   SET status = 'BUSY',
                       current_task_id = $1,
                       last_active_at = $2
                   WHERE id = $3""",
                task_id,
                datetime.now(),
                claim_request.agent_id
            )
            
            # 6. 记录活动日志
            await conn.execute(
                """INSERT INTO activity_logs 
                   (agent_id, action, task_id, details)
                   VALUES ($1, $2, $3, $4)""",
                claim_request.agent_id,
                'claim',
                task_id,
                json.dumps({"task_title": task['title']})
            )
            
            return {
                "success": True,
                "task": dict(task)
            }
```

### Phase 3: Agent 唤醒逻辑改造 (1 天)

```javascript
const API_BASE = 'http://localhost:6565/api';

async function wakeUp() {
  console.log(`[${this.agentId}] 唤醒...`);
  
  try {
    // 1. 获取可用任务
    const availableTasks = await this.getAvailableTasks();
    
    if (availableTasks.length === 0) {
      console.log(`[${this.agentId}] 无可用任务`);
      return;
    }
    
    // 2. 选择适合的任务
    const suitableTask = this.selectTask(availableTasks);
    
    if (!suitableTask) {
      console.log(`[${this.agentId}] 没有适合的任务`);
      return;
    }
    
    // 3. 尝试认领
    const claimResult = await this.claimTask(suitableTask.id);
    
    if (!claimResult.success) {
      console.log(`[${this.agentId}] 认领失败：${claimResult.error}`);
      return;
    }
    
    // 4. 执行任务
    console.log(`[${this.agentId}] 开始执行任务：${suitableTask.title}`);
    await this.executeTask(suitableTask);
    
    // 5. 提交交付物
    await this.submitDeliverable(suitableTask.id);
    
  } catch (error) {
    console.error(`[${this.agentId}] 唤醒失败:`, error);
  }
}

async function getAvailableTasks() {
  const response = await fetch(`${API_BASE}/tasks/available`);
  const data = await response.json();
  
  if (data.success) {
    return data.data;
  } else {
    return [];
  }
}

function selectTask(tasks) {
  // 1. 按角色筛选
  const suitable = tasks.filter(task => 
    this.matchRole(task, this.role)
  );
  
  // 2. 按优先级排序
  suitable.sort((a, b) => {
    const priorityOrder = { 'P0': 0, 'P1': 1, 'P2': 2, 'P3': 3 };
    return priorityOrder[a.priority] - priorityOrder[b.priority];
  });
  
  // 3. 返回最高优先级的任务
  return suitable[0];
}

async function claimTask(taskId) {
  const response = await fetch(`${API_BASE}/tasks/${taskId}/claim`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      agent_id: this.agentId
    })
  });
  
  return await response.json();
}
```

### Phase 4: 数据迁移工具 (0.5 天)

#### 4.1 文件→数据库迁移脚本

**文件**: `/home/admin/.openclaw/workspace/task-board/scripts/migrate-to-db.js`

```javascript
const fs = require('fs');
const path = require('path');
const sqlite3 = require('sqlite3').verbose();

const TASKS_DIR = '/home/admin/.openclaw/workspace/task-board/tasks';
const DB_PATH = '/home/admin/.openclaw/workspace/task-board/database/task-board.db';

async function migrate() {
  const db = new sqlite3.Database(DB_PATH);
  
  // 读取所有任务文件
  const files = fs.readdirSync(TASKS_DIR).filter(f => f.endsWith('.md'));
  
  for (const file of files) {
    const filePath = path.join(TASKS_DIR, file);
    const content = fs.readFileSync(filePath, 'utf8');
    
    // 解析任务文件
    const task = parseTaskFile(content, file);
    
    // 插入数据库
    await db.run(
      `INSERT OR REPLACE INTO tasks 
       (id, title, priority, status, assignee, description, created_at)
       VALUES (?, ?, ?, ?, ?, ?, ?)`,
      [
        task.taskId,
        task.taskName,
        task.priority,
        task.status,
        task.assignee,
        task.description,
        new Date()
      ]
    );
    
    console.log(`✅ 迁移任务：${task.taskId}`);
  }
  
  db.close();
  console.log('✅ 迁移完成');
}

function parseTaskFile(content, filename) {
  // 解析 Markdown 任务文件
  const taskId = filename.replace('.md', '');
  const taskName = content.match(/# (.+)/)?.[1] || taskId;
  const priority = content.match(/\*\*优先级\*\*:\s*(P[0-3])/)?.[1] || 'P2';
  const status = content.match(/\*\*状态\*\*:\s*\w+ (\w+)/)?.[1] || 'PENDING';
  const assignee = content.match(/\*\*执行者\*\*:\s*(\S+)/)?.[1] || null;
  const description = content;
  
  return { taskId, taskName, priority, status, assignee, description };
}

migrate();
```

### Phase 5: 超时释放机制 (0.5 天)

#### 5.1 定时清理脚本

**文件**: `/home/admin/.openclaw/workspace/task-board/scripts/release-timeout-tasks.js`

```javascript
const sqlite3 = require('sqlite3').verbose();

const DB_PATH = '/home/admin/.openclaw/workspace/task-board/database/task-board.db';
const TIMEOUT_MINUTES = 30; // 30 分钟超时

async function releaseTimeoutTasks() {
  const db = new sqlite3.Database(DB_PATH);
  
  const timeout = new Date(Date.now() - TIMEOUT_MINUTES * 60 * 1000);
  
  // 查找超时任务
  const tasks = await db.all(
    `SELECT * FROM tasks 
     WHERE status = 'ASSIGNED' 
       AND claimed_at < ?`,
    [timeout]
  );
  
  for (const task of tasks) {
    console.log(`⚠️ 任务 ${task.id} 超时，自动释放`);
    
    // 释放任务
    await db.run(
      `UPDATE tasks 
       SET status = 'AVAILABLE',
           claimed_by = NULL,
           claimed_at = NULL,
           updated_at = CURRENT_TIMESTAMP
       WHERE id = ?`,
      [task.id]
    );
    
    // 更新 Agent 状态
    await db.run(
      `UPDATE agents 
       SET status = 'IDLE',
           current_task_id = NULL
       WHERE id = ?`,
      [task.claimed_by]
    );
    
    // 记录日志
    await db.run(
      `INSERT INTO activity_logs (agent_id, action, task_id, details)
       VALUES (?, 'timeout_release', ?, ?)`,
      [task.claimed_by, task.id, JSON.stringify({ reason: 'timeout' })]
    );
  }
  
  db.close();
  console.log(`✅ 已释放 ${tasks.length} 个超时任务`);
}

releaseTimeoutTasks();
```
